function Editor(input, preview) {
        this.update = function () {
            preview.innerHTML = markdown.toHTML(input.value);
        };
        input.editor = this;
        this.update();
        $("pre code").each(function (i, block) {
            console.log("已经渲染第" + i + "个");
            hljs.highlightBlock(block);
        });
    }
